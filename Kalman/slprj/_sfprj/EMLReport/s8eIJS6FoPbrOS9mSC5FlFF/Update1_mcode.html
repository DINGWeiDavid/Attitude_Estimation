1%% EKF - Update           %
  2% Author: Mattia Giurato  %
  3% Last review: 2016/03/15 %
  4%%%%%%%%%%%%%%%%%%%%%%%%%%%
  5function [Pkp,qkp,omekhat,RPYkal,betakp] = Update(Pkm,qkm,betakm,Gyroscope,Accelerometer,Magnetometer)
  6% Usefull values
  7sigma_acc = diag([0.77418 0.66936 0.71485]);
  8sigma_mag = 300*diag([0.053667 0.034382 0.071054]);
  9% Initialization of the output values
 10Pkp = eye(6);
 11betakp = zeros(3,1);
 12qkp = zeros(4,1);
 13omekhat = zeros(3,1);
 14RPYkal = zeros(3,1);
 15
 16% Normalise accelerometer measurement
 17if(norm(Accelerometer) == 0), return; end % handle NaN
 18b_acc = Accelerometer / norm(Accelerometer); % normalise magnitude
 19
 20% Normalise magnetometer measurement
 21if(norm(Magnetometer) == 0), return; end % handle NaN
 22b_mag = Magnetometer / norm(Magnetometer); % normalise magnitude
 23
 24%% Propagated value from previous step
 25deltaXkm = zeros(6,1);
 26
 27  %% Accelerometer and Magnetometer correction     
 28  for i = 1:1:2
 29      % Compute attitude matrix
 30      ro = [qkm(1) ;
 31            qkm(2) ;
 32            qkm(3)];
 33      q_4 = qkm(4);
 34      rox = [   0   -ro(3)  ro(2) ;
 35               ro(3)  0    -ro(1) ;
 36              -ro(2) ro(1)   0   ];
 37      Aqkm = (q_4^2 - norm(ro)^2)*eye(3) + 2*(ro*ro') - 2*q_4*rox;
 38      
 39      if i == 1
 40          % Reference direction of Earth's gravitational field
 41          r_acc = [0 ;
 42                   0 ;
 43                   1];
 44          Aqmr = Aqkm*r_acc;
 45          b = b_acc;
 46          R = sigma_acc^2*eye(3);
 47          
 48          % Sensitivity Matrix
 49          Aqmrx = [      0   -Aqmr(3)  Aqmr(2) ;
 50                     Aqmr(3)    0     -Aqmr(1) ;
 51                    -Aqmr(2)  Aqmr(1)    0    ];
 52          Hk = [Aqmrx zeros(3)];
 53          
 54          % Gain
 55          Kk = (Pkm*Hk')/(Hk*Pkm*Hk' + R);
 56          
 57          % Update Covariance
 58          Pkp = (eye(6) - Kk*Hk)*Pkm*(eye(6) - Kk*Hk)' + Kk*R*Kk';
 59          
 60          % Update state
 61          epk = b - Aqmr; %residual
 62          yk = epk - Hk*deltaXkm;
 63          deltaXkp = diag([1 1 0 1 1 0])*Kk*yk;
 64          
 65          % Update quaternion
 66          dalphak = deltaXkp(1:3,1);
 67          dq = (1/sqrt(4+norm(dalphak)^2))*[dalphak' 2]';
 68          qkp = [dq(4)*ro + q_4*dq(1:3,1)-cross(dq(1:3,1),ro) ;
 69                         dq(4)*q_4 - dot(dq(1:3,1),ro)       ];
 70          
 71          % Update biases
 72          deltaBeta = deltaXkp(4:6,1);
 73          betakp = betakm + deltaBeta;
 74          
 75          Pkm = Pkp;
 76          deltaXkm = deltaXkp;
 77          betakm = betakp;
 78          qkm = qkp;
 79          
 80      elseif i == 2
 81          % Reference direction of Earth's magnetic feild
 82          h = Aqkm*Magnetometer;
 83          r_mag = [norm([h(1) h(2)]) ;
 84              0          ;
 85              h(3)       ];
 86          Aqmr = Aqkm*r_mag;
 87          b = b_mag;
 88          R = sigma_mag^2*eye(3);
 89          
 90          % Sensitivity Matrix
 91          Aqmrx = [      0   -Aqmr(3)  Aqmr(2) ;
 92                     Aqmr(3)    0     -Aqmr(1) ;
 93                    -Aqmr(2)  Aqmr(1)    0    ];
 94          Hk = [Aqmrx zeros(3)];
 95          
 96          % Gain
 97          Kk = (Pkm*Hk')/(Hk*Pkm*Hk' + R);
 98          
 99          % Update Covariance
100          Pkp = (eye(6) - Kk*Hk)*Pkm*(eye(6) - Kk*Hk)' + Kk*R*Kk';
101          
102          % Update state
103          epk = b - Aqmr; %residual
104          yk = epk - Hk*deltaXkm;
105          deltaXkp = diag([0 0 1 0 0 1])*(deltaXkm + Kk*yk);
106          
107          % Update quaternion
108          dalphak = deltaXkp(1:3,1);
109          dq = (1/sqrt(4+norm(dalphak)^2))*[dalphak' 2]';
110          qkp = [dq(4)*ro + q_4*dq(1:3,1)-cross(dq(1:3,1),ro) ;
111                         dq(4)*q_4 - dot(dq(1:3,1),ro)       ];
112          
113          % Update biases
114          deltaBeta = deltaXkp(4:6,1);
115          betakp = betakm + deltaBeta;
116      end
117  end
118%% Depolarize gyroscope measurement
119omekhat = Gyroscope - betakp;
120
121%% Convert Quaternion into RPY
122a1 = 2.*(qkp(2).*qkp(3) + qkp(1).*qkp(4));
123a2 = qkp(1).^2 + qkp(2).^2 - qkp(3).^2 - qkp(4).^2;
124b = -2.*(qkp(2).*qkp(4) - qkp(1).*qkp(3));
125c1 = 2.*(qkp(3).*qkp(4) + qkp(1).*qkp(2));
126c2 = qkp(1).^2 - qkp(2).^2 - qkp(3).^2 + qkp(4).^2;
127
128R = atan2( a1, a2 );
129P = asin( b );
130Y = atan2( c1, c2 );
131
132RPYkal = [R -P -Y];
133
